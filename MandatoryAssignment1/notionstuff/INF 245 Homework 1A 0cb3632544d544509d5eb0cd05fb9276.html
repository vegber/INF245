def JacobiSymbol(a, n, d=1):
    if a == 0:
        return 0  # (0/n) = 0
    if a == 1:
        return d  # (1/n) = 1

    if a < 0:
        # property of Jacobi
        # (a/n) = (-a/n)*(-1/n)
        a = -a
        if n % 4 == 3:
            # (-1/n) = -1 if n = 3 (mod 4)
            d = -d

    while a:
        if a < 0:
            # (a/n) = (-a/n)*(-1/n)
            a = -a
            if n % 4 == 3:
                # (-1/n) = -1 if n = 3 (mod 4)
                d = -d
        while a % 2 == 0:
            a = a // 2
            if n % 8 == 3 or n % 8 == 5:
                d = -d
        # swap
        a, n = n, a
        if a % 4 == 3 and n % 4 == 3:
            d = -d
        a = a % n
        if a > n // 2:
            a = a - n
    if n == 1:
        return d
    return 0

def Solovay_Strassen_Test(n, k=20) -> str:
    """
    :input: n, a value to test primality
    :out: composite if test fails, probably prime else
    :rtype: str
    """
    # check n odd prime > 1
    assert (n > 1)
    import math
    for i in range(k):
        # Solovay strassen test:
        # 1) gcd(a, n) != 1 => composite
